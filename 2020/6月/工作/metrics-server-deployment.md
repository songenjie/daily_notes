  name: metrics-server

apiVersion: extensions/v1beta1
kind: Deployment
metadata:
  name: metrics-server
  namespace: kube-system
  labels:
    k8s-app: metrics-server
spec:
  selector:
    matchLabels:
      k8s-app: metrics-server
  template:
    metadata:
      name: metrics-server
      labels:
        k8s-app: metrics-server
    spec:
      hostNetwork: true
      serviceAccountName: metrics-server
      volumes:
      # mount in tmp so we can safely use from-scratch images and/or read-only containers
      - name: tmp-dir
        emptyDir: {}
      containers:
      - name: metrics-server
        image: images name
        command: ["/metrics-server","--kubelet-insecure-tls","--secure-port=$PORT","-v","6"]
        imagePullPolicy: Always
        ports:
        - containerPort: $PORT
          protocol: TCP
        volumeMounts:
        - name: tmp-dir
          mountPath: /tmp
